10import java.util.*;
public class StackWithTwoQueues {
	private Queue<Integer> queue1 = new ArrayDeque<>();
	private int size;
	
	public void push(int value){
		queue1.add(value);
		size++;
	}
	
	public int size(){
		return size;
	}
	
	public int pop(){
		if(isEmpty()){
			System.out.println("Stack is Empty");
			return -1;
		}
		else{
			int value = queue1.remove();
			size--;
			return value;
		}
	}
	
	public int peek(){
		if(isEmpty()){
			System.out.println("Stack is empty");
			return -1;
		}
		else{
			return queue1.peek();
		}
	}
	
	public boolean isEmpty(){
		return size == 0;
	}
	
	public void print(){
		while(!queue1.isEmpty()){
			System.out.print(queue1.remove()+" ");
		}
		size = 0;
		System.out.println();
	}
}
